import os import google.generativeai as genai # 1. 設定（環境変数などから取得） genai.configure(api_key=os.environ["GEMINI_API_KEY"]) model = genai.GenerativeModel('gemini-1.5-flash') def generate_article(topic): # 2. 「国民のため」と「利益」を両立させるプロンプト prompt = f""" テーマ: {topic} について、読者に価値のあるブログ記事を日本語で作成してください。 【構成ルール】 1. 読者のメリット: その商品や情報の「利点」と「注意点」を客観的に書く。 2. 専門性: 詳しいスペックや比較情報を含める。 3. 収益性: 記事の自然な流れの中で、関連アイテムの紹介を入れる。 4. デザイン: シンプルで見やすく、テキストは黒（#000000）で統一。 HTML形式の
タグのみで出力してください。 """ response = model.generate_content(prompt) return response.text def update_html(new_content): # 3. 既存のcherry.htmlに追記、または新しいファイルを作成 file_path = "cherry.html" # 既存のファイルを読み込み、新しいコンテンツを上部に追加するロジック with open(file_path, "r", encoding="utf-8") as f: soup = f.read() # 特定のタグ（例: ）の直後に挿入 updated_soup = soup.replace("", f"\n{new_content}") with open(file_path, "w", encoding="utf-8") as f: f.write(updated_soup) # 実行 new_article = generate_article("ハーベスト・チェリーと合うコーヒーの組み合わせ") update_html(new_article)
